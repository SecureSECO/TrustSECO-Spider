 18class Controller:
 19    """
 20    Class to facilitate input from, and output to, the Node.JS program.
 21
 22    It will try and get input from the console in form of a JSON string.
 23    It will then go through this JSON object to find the required information, and the wanted data-points.
 24    These data-points will then be requested from the actual API classes,
 25    and then returned to the Node.JS program by placing a JSON string on the console
 26    """
 27
 28    def __init__(self) -> None:
 29        # API objects
 30        self.gh_api = GitHubAPICall()
 31        self.lib_api = LibrariesAPICall()
 32
 33        # Spider objects
 34        self.gh_spider = GitHubSpider()
 35        self.cve_spider = CVESpider()
 36
 37    def run(self, input_json):
 38        """
 39        This is the main looping function of the program.
 40
 41        It will try to read the console to see if a new command has been received.
 42        """
 43
 44        # Make sure we got the information we need
 45        if 'project_info' not in input_json:
 46            print('Error: no project information found')
 47            return 'Error: no project information found'
 48
 49        # Make sure all of the wanted project information is available
 50        if 'project_platform' and 'project_owner' and 'project_name' and 'project_release' and 'project_year' in input_json["project_info"]:
 51            # Retrieve the project information
 52            platform = input_json["project_info"]["project_platform"]
 53            owner = input_json["project_info"]["project_owner"]
 54            repo_name = input_json["project_info"]["project_name"]
 55            release = input_json["project_info"]["project_release"]
 56            year = input_json["project_info"]["project_year"]
 57
 58            # Create an output JSON object
 59            output_json = {}
 60
 61            # Request the data from GitHub
 62            if 'gh_data_points' in input_json:
 63                # Tell the user what is going on
 64                print('-------------------')
 65                print('Getting GitHub data...')
 66
 67                # Actually request the data
 68                output_json.update({'gh_data_points': self.get_github_data(
 69                    owner, repo_name, release, year, input_json["gh_data_points"])})
 70
 71            # Request the data from Libraries.IO
 72            if 'lib_data_points' in input_json:
 73                # Tell the user what is going on
 74                print('-------------------')
 75                print('Getting Libraries.IO data...')
 76
 77                # Libraries.io does not use 'v' in their version numbers, so we need to remove it if it is there
 78                if release[0].lower() == 'v':
 79                    lib_release = release[1:]
 80                else:
 81                    lib_release = release
 82
 83                # Actually request the data
 84                output_json.update({'lib_data_points': self.get_libraries_data(
 85                    platform, owner, repo_name, lib_release, input_json["lib_data_points"])})
 86
 87            # Request the data from the CVE website
 88            if 'cve_data_points' in input_json:
 89                # Tell the user what is going on
 90                print('-------------------')
 91                print('Getting CVE data...')
 92
 93                # Actually request the data
 94                output_json.update({'cve_data_points': self.get_cve_data(
 95                    repo_name, input_json["cve_data_points"])})
 96
 97            # Print the output JSON object to the console
 98            return output_json
 99        else:
100            print('Error: missing project information')
101            return 'Error: missing project information'

103    def get_github_data(self, owner, repo_name, release, year, wanted_data):
104        """
105        This function will get the data from GitHub.
106
107        It will then return a JSON string containing the data.
108        """
109
110        # Create a JSON object to store the data
111        return_data = {}
112
113        # Loop through the wanted data list and retrieve the data
114        for data_point in wanted_data:
115            if data_point == "gh_contributor_count":
116                return_data.update(
117                    {data_point: self.gh_api.get_repository_contributor_count(owner, repo_name)})
118            elif data_point == "gh_user_count":
119                return_data.update(
120                    {data_point: self.gh_spider.get_repository_user_count(owner, repo_name)})
121            elif data_point == "gh_total_download_count":
122                return_data.update(
123                    {data_point: self.gh_api.get_total_download_count(owner, repo_name)})
124            elif data_point == "gh_release_download_count":
125                return_data.update(
126                    {data_point: self.gh_api.get_release_download_count(owner, repo_name, release)})
127            elif data_point == "gh_yearly_commit_count":
128                return_data.update(
129                    {data_point: self.gh_api.get_yearly_commit_count(owner, repo_name)})
130            elif data_point == "gh_given_year_commit_count":
131                return_data.update(
132                    {data_point: self.gh_api.get_commit_count_in_year(owner, repo_name, year)})
133            elif data_point == "gh_repository_language":
134                return_data.update(
135                    {data_point: self.gh_api.get_repository_language(owner, repo_name)})
136            elif data_point == "gh_gitstar_ranking":
137                return_data.update(
138                    {data_point: self.gh_api.get_gitstar_ranking(owner, repo_name)})
139            elif data_point == "gh_open_issues_count":
140                return_data.update(
141                    {data_point: self.gh_spider.get_repository_open_issue_count(owner, repo_name)})
142            elif data_point == "gh_zero_response_issues_count":
143                return_data.update(
144                    {data_point: self.gh_api.get_zero_responses_issue_count(owner, repo_name)})
145            elif data_point == "gh_release_issues_count":
146                return_data.update(
147                    {data_point: self.gh_api.issue_count_per_release(owner, repo_name, release)})
148            elif data_point == "gh_issue_ratio":
149                return_data.update(
150                    {data_point: self.gh_spider.get_repository_issue_ratio(owner, repo_name)})
151            elif data_point == "gh_average_resolution_time":
152                return_data.update(
153                    {data_point: self.gh_api.get_average_issue_resolution_time(owner, repo_name)})
154            elif data_point == "gh_owner_stargazer_count":
155                return_data.update(
156                    {data_point: self.gh_api.get_owner_stargazer_count(owner)})
157            else:
158                print(f"Error: invalid data point {data_point}")
159                return_data.update({data_point: None})
160
161        # Return the requested data-points
162        return return_data

248    """
249    This function will update the environmental variables with the given GitHub token
250    """
251
252    # Make sure the .env file exists
253    if not os.path.exists(constants.ENVIRON_FILE):

256
257    # Update the .env file
258    set_key(constants.ENVIRON_FILE, constants.GITHUB_TOKEN, github_token)
